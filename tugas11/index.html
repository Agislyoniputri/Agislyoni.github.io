01 LATIHAN: CIRCLE REDIRECTING CONSTRUCTORS

Mengimplementasikan konsep Redirecting Constructors pada kelas Circle. Konstruktor utama menerima parameter radius yang dilengkapi dengan validasi input ketat, sedangkan tiga konstruktor pengarah (fromDiameter, fromKeliling, dan fromLuas) mengalihkan instansiasi ke konstruktor utama tanpa memiliki body terpisah (menerapkan prinsip DRY - Don't Repeat Yourself).

import 'dart:math' as math;

class Circle {
  final double radius;

  Circle(this.radius) {
    if (radius <= 0) {
      throw ArgumentError('💫 Jari-jari lingkaran wajib bernilai positif! 🦋');
    }
  }

  Circle.fromDiameter(double d) : this(d / 2.0);

  Circle.fromKeliling(double circumference) : this(circumference / (2.0 * math.pi));

  Circle.fromLuas(double area) : this(math.sqrt(area / math.pi));

  double get diameter => radius + radius;
  double get keliling => 2.0 * math.pi * radius;
  double get luas => math.pi * math.pow(radius, 2);

  @override
  String toString() {
    return 'Circle(radius: ${radius.toStringAsFixed(2)}, '
        'diameter: ${diameter.toStringAsFixed(2)}, '
        'keliling: ${keliling.toStringAsFixed(2)}, '
        'luas: ${luas.toStringAsFixed(2)})';
  }
}

import 'libcircle.dart';

void main() {
  print('📌 DEMO CONSTRUCTOR PENGARAH KELAS CIRCLE - AGIS LYONI 📌\n');

  print('🔹 [Circle 1] Inisialisasi Melalui Radius (Main Constructor: 5.0) 🔹');
  final circle1 = Circle(5.0);
  print('🌸 $circle1');
  print('');

  print('🔹 [Circle 2] Inisialisasi Melalui Diameter (Redirecting: 10.0) 🔹');
  final circle2 = Circle.fromDiameter(10.0);
  print('🌸 $circle2');
  print('✨ Apakah radius circle2 sama dengan circle1? ${circle1.radius == circle2.radius}');
  print('');

  print('🔹 [Circle 3] Inisialisasi Melalui Keliling (Redirecting) 🔹');
  final circle3 = Circle.fromKeliling(31.41592653589793);
  print('🌸 $circle3');
  print('✨ Apakah radius circle3 mendekati circle1? ${(circle1.radius - circle3.radius).abs() < 1e-9}');
  print('');

  print('🔹 [Circle 4] Inisialisasi Melalui Luas (Redirecting) 🔹');
  final circle4 = Circle.fromLuas(78.53981633974483);
  print('🌸 $circle4');
  print('✨ Apakah radius circle4 mendekati circle1? ${(circle1.radius - circle4.radius).abs() < 1e-9}');
  print('');

  print('🔸 Uji Validasi Radius Negatif (Ekspektasi Error) 🔸');
  try {
    final invalidCircle = Circle(-2.0);
    print(invalidCircle);
  } catch (e) {
    print('🌸 ERROR DITANGKAP: $e');
  }
  print('');

  print('🔸 Uji Validasi Diameter Nol (Ekspektasi Error) 🔸');
  try {
    final invalidDiameter = Circle.fromDiameter(0);
    print(invalidDiameter);
  } catch (e) {
    print('🌸 ERROR DITANGKAP: $e');
  }
  print('');

  print('🔸 Uji Validasi Keliling Negatif (Ekspektasi Error) 🔸');
  try {
    final invalidKeliling = Circle.fromKeliling(-10.0);
    print(invalidKeliling);
  } catch (e) {
    print('🌸 ERROR DITANGKAP: $e');
  }
  print('');

  print('🔸 Uji Validasi Luas Negatif (Ekspektasi Error) 🔸');
  try {
    final invalidLuas = Circle.fromLuas(-50.0);
    print(invalidLuas);
  } catch (e) {
    print('🌸 ERROR DITANGKAP: $e');
  }
}
